
Endpoints:

GET /rides
GET /rides/{id}
POST /rides
GET /rides/{id}/pickups
POST /rides/{id}/pickups
PUT /rides/{id}/pickups
GET /profiles/{id}


Models:

class Location:
	latitude: float
	longitude: float


class RideSlim:
	driver_id: uuid
	image: str
	event_time: datetime
	distance: float
	ride_id: uuid
	location: Location
	seats: int
	to_uni: bool

class Ride:
	driver_id: uuid
	name: str
	image: str
	event_time: datetime
	distance: float #  dynamic
	ride_id: uuid
	location: Location
	seats: int
	to_uni: bool
	passengers: List[Profile]
	price: float
	description: str
	request_status: Enum #  dynamic


class Profile:
	profile_id: uuid
	score: float
	name: str
	interests: List[str]
	education: str
	city: str
	bio: str
	gender: str
	image: str


class Pickup:
	profile_id: uuid
	status: Enum
	ride_id: uuid
	name: str
	image: str
